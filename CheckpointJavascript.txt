Exercice 8
Le minimum dans un tableau

var t = [3,4,7,2]

var min = t[0]

for(var i = 0; i<t.length; i++){
  if(t[i]<min){
    min = t[i]
  }
}

console.log(min)


Exercice 9
Le nombre d'occurrences d'un nombre dans un tableau

var t = [3,4,4,5,3,8,7,3]
var num = 3
var count = 0

for(var i = 0; i<t.length;i++){
  if(t[i] == num){
    count ++
  }
}

console.log(count)
